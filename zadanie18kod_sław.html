var a = parseFloat(prompt("Podaj liczbę a:", ""));
    var b = parseFloat(prompt("Podaj liczbę b:", ""));

    function suma_sla(a, b) {
        return a + b;
    }

    function odejmowanie_sla(a, b) {
        return a - b;
    }

    function mnozenie_sla(a, b) {
        return a * b;
    }

    function od_kwadratu_sla(a, b) {
        return "a^2 = " + (a ** 2) + ", b^2 = " + (b ** 2);
    }

    function reszta_z_dzielenia_a_przez_b_sla(a, b) {
        return a % b;
    }

    document.write("Suma: " + suma_sla(a, b) + "
");
    document.write("Różnica: " + odejmowanie_sla(a, b) + "
");
    document.write("Iloczyn: " + mnozenie_sla(a, b) + "
");
    document.write("Kwadraty: " + od_kwadratu_sla(a, b) + "
");
    document.write("Reszta z dzielenia a przez b: " + reszta_z_dzielenia_a_przez_b_sla(a, b) + "
");
