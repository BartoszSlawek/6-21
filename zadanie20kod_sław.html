function AB_sła() {
        var xa_sła = parseFloat(document.getElementById('xa').value);
        var ya_sła = parseFloat(document.getElementById('ya').value);
        var xb_sła = parseFloat(document.getElementById('xb').value);
        var yb_sła = parseFloat(document.getElementById('yb').value);
        var ab = Math.sqrt(Math.pow(xb_sła - xa_sła, 2) + Math.pow(yb_sła - ya_sła, 2));
        document.getElementById("wynik").innerHTML = "Długość AB = " + ab.toFixed(2) + " cm";
    }

    function BC_sła() {
        var xb_sła = parseFloat(document.getElementById('xb').value);
        var yb_sła = parseFloat(document.getElementById('yb').value);
        var xc_sła = parseFloat(document.getElementById('xc').value);
        var yc_sła = parseFloat(document.getElementById('yc').value);
        var bc = Math.sqrt(Math.pow(xc_sła - xb_sła, 2) + Math.pow(yc_sła - yb_sła, 2));
        document.getElementById("wynik").innerHTML = "Długość BC = " + bc.toFixed(2) + " cm";
    }

    function AC_sła() {
        var xa_sła = parseFloat(document.getElementById('xa').value);

    }
