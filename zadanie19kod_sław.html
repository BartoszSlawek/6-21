function pole_sławek() {
        var a_sła = parseFloat(document.getElementById('dlugosc').value);
        var b_sła = a_sła * a_sła;
        document.getElementById("nazwa").innerHTML = "Pole kwadratu= " + b_sła.toFixed(2) + " cm²";
    }

    function obwod_sławek() {
        var a_sła = parseFloat(document.getElementById('dlugosc').value);
        var b_sła = 4 * a_sła;
        document.getElementById("nazwa").innerHTML = "Obwód kwadratu= " + b_sła.toFixed(2) + " cm";
    }
